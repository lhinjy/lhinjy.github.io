Chimecho and Flying Cat

math FVRML Parametric Equations 3D Modeling

The models are pieced together by individual shapes. Individual components are created through their parametric equations.


Chimecho

FShape {
geometry FGeometry {
	definition "function frep(x,y,z,t){
	head = 0.4^2 - (x)^2 - (y)^2 - z^2;
    top = 1 - (x/0.25)^2 - ((y-0.6)/0.1)^2 - (z/0.15)^2;
    bodyline = min( 1 -(z/0.05)^2 -(x/0.05)^2, min(0.5-y,y+1.5));
    tail = min((((z+0.6)/0.8)^2 - (x/0.5)^2 - ((y+0.075)/0.5)^2),-0.1-z);
    topbody = max((max(max(head,top),bodyline)),tail);

	conebelow =min((((y+0.5)/1)^2 -(x/1)^2 - (z/0.4)^2),-y-0.5);
	rectangle = min(  min (min(x+0.25,-x+0.25),min(-y-0.5,y+1.5) ), min(z+0.05,0.05-z) );
	bodybot = min(conebelow,rectangle);

	body = max(topbody, bodybot);

	eye1 = 0.06^2 - (x-0.2)^2 - y^2 - (z-0.38)^2;
	eye2 = 0.06^2 - (x+0.2)^2 - y^2 - (z-0.38)^2;
	eye = max(eye1,eye2);
	hand = 1 - (x/0.5)^2 -((y+0.1)/0.1)^2 -((z-0.15)/0.1)^2;

	final = max(min(body,-eye),hand);
    return final;
    }"
bboxCenter 0 -0.5 -0.1
bboxSize  1 2.5 1
resolution [90 95 90]}
appearance FAppearance {
material FMaterial {
# Variable color is defined for the CGS solid
diffuseColor "r=(w+1)/2; g=abs(cos(v*pi/2)); b=0.9;"}}
}

Flying Cat

FShape {
geometry FGeometry {

# Function script defining the CSG solid.
# Change to some other formulae to see how the solid geometry changes
# based on the parameters domain and the sampling resolution defined below
	definition "function frep(x,y,z,t){
    sphere = 0.5^2 - (x+1)^2 - y^2 - z^2;
    mouth =  min(min(min(-0.11-y,y+0.12),-x),min(z+0.15,0.15-z));

    eye1 = 0.05^2 - (x+1.44)^2 - (y)^2 - (z-0.2)^2;
    eye2 = 0.05^2 - (x+1.44)^2 - (y)^2 - (z+0.2)^2;
    eye = max(eye1,eye2);

	head = min(min(sphere,-mouth),-eye);

	bodyA = min( 1-((y+0.15)/0.3)^2-(z/0.3)^2, min (x+1, -x));
    bodyB =0.30^2 - x^2 -(y+0.15)^2 -z^2;
	body = max(bodyA,bodyB);

    feetp = 1 - ((x-0.2)/0.5)^2 - ((y+0.3)/0.18)^2 - ((z-0.2)/0.18)^2 ;
    feetn = 1 - ((x-0.2)/0.5)^2 - ((y+0.3)/0.18)^2 - ((z+0.2)/0.18)^2 ;
    feet= max(feetp , feetn);

    hand = 1 - ((x+0.5)/0.2)^2 - ((y+0.25)/0.2)^2 - (z/0.7)^2;

    ear1 =min( ((y-1)/3)^2 - ((x+1)/0.3)^2 - ((z-0.2))^2, y-0.2);
    ear2 = min( ((y-1)/3)^2 - ((x+1)/0.3)^2 - ((z+0.2))^2, y-0.2);
    ear  = max (ear1, ear2);

    top = min(max(ear,head),-eye);
    bot = max(max(body,feet),hand);

    cape = min(min(min(y-0.5*x -0.3, 0.5*x +0.34-y),min(z+0.28,0.28-z)),x+0.55);


    final = max(max(top,bot),cape);
    return final;}"

# Adjust the tight bounding box and an optimal resolution
bboxCenter -0.8 0.2 0
bboxSize 3 1.6 1.5
resolution [225 225 225]
}

appearance FAppearance {
material FMaterial {
# Variable color is defined for the CGS solid
diffuseColor "r=abs(cos(2*pi*w*v)); g=(v+1)/2; b=0.7;"}}
}
